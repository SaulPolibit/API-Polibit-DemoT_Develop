# PDF Generator Patterns

## PDFKit Setup

All PDF generators use `pdfkit` (v0.17.2) with a consistent pattern:

```javascript
const PDFDocument = require('pdfkit');

async function generateReportPDF(data) {
  return new Promise((resolve, reject) => {
    const doc = new PDFDocument({
      size: 'A4',
      margins: { top: 50, bottom: 50, left: 50, right: 50 },
      info: {
        Title: 'Report Title',
        Author: 'Polibit',
      },
    });

    const buffers = [];
    doc.on('data', chunk => buffers.push(chunk));
    doc.on('end', () => resolve(Buffer.concat(buffers)));
    doc.on('error', reject);

    // Add content...
    addHeader(doc, data);
    addContent(doc, data);
    addFooter(doc);

    doc.end();
  });
}
```

## Common Helper Functions

### Header
```javascript
function addHeader(doc, { fundName, reportTitle, reportDate }) {
  doc.fontSize(20)
     .fillColor(COLORS.primary)
     .text(fundName, 50, 50);

  doc.fontSize(14)
     .fillColor(COLORS.textLight)
     .text(reportTitle, 50, 75);

  doc.fontSize(10)
     .text(`Generated: ${formatDate(reportDate)}`, 50, 95);

  // Divider line
  doc.moveTo(50, 115)
     .lineTo(545, 115)
     .strokeColor(COLORS.border)
     .stroke();

  doc.moveDown(2);
}
```

### Section Title
```javascript
function addSectionTitle(doc, title) {
  doc.fontSize(14)
     .fillColor(COLORS.primary)
     .text(title);

  doc.moveTo(doc.x, doc.y + 2)
     .lineTo(doc.x + 200, doc.y + 2)
     .strokeColor(COLORS.secondary)
     .stroke();

  doc.moveDown(0.5);
}
```

### Data Table
```javascript
function addTable(doc, { headers, rows, columnWidths }) {
  const startX = 50;
  const startY = doc.y;
  const rowHeight = 20;

  // Header row (with background)
  doc.rect(startX, startY, 495, rowHeight)
     .fill(COLORS.primary);

  headers.forEach((header, i) => {
    const x = startX + sumWidths(columnWidths, i);
    doc.fontSize(9)
       .fillColor(COLORS.white)
       .text(header, x + 5, startY + 5, { width: columnWidths[i] - 10 });
  });

  // Data rows
  rows.forEach((row, rowIndex) => {
    const y = startY + (rowIndex + 1) * rowHeight;

    // Alternating row background
    if (rowIndex % 2 === 0) {
      doc.rect(startX, y, 495, rowHeight)
         .fill(COLORS.background);
    }

    row.forEach((cell, colIndex) => {
      const x = startX + sumWidths(columnWidths, colIndex);
      doc.fontSize(9)
         .fillColor(COLORS.text)
         .text(cell, x + 5, y + 5, { width: columnWidths[colIndex] - 10 });
    });
  });

  doc.y = startY + (rows.length + 1) * rowHeight + 10;
}
```

### Footer with Page Numbers
```javascript
function addFooter(doc) {
  const pages = doc.bufferedPageRange();
  for (let i = pages.start; i < pages.start + pages.count; i++) {
    doc.switchToPage(i);
    doc.fontSize(8)
       .fillColor(COLORS.textLight)
       .text(
         `Page ${i + 1} of ${pages.count}`,
         50, doc.page.height - 50,
         { align: 'center' }
       );

    doc.text(
      'Confidential — Generated by Polibit',
      50, doc.page.height - 35,
      { align: 'center' }
    );
  }
}
```

## Formatting Utilities

```javascript
// Currency formatting
function formatCurrency(amount, currency = 'USD') {
  return new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency,
    minimumFractionDigits: 2,
  }).format(amount);
}

// Percentage formatting
function formatPercentage(value, decimals = 2) {
  return `${(value * 100).toFixed(decimals)}%`;
}

// Date formatting
function formatDate(date) {
  return new Date(date).toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
  });
}

// Number formatting
function formatNumber(value, decimals = 0) {
  return new Intl.NumberFormat('en-US', {
    minimumFractionDigits: decimals,
    maximumFractionDigits: decimals,
  }).format(value);
}
```

## Page Break Handling

```javascript
function checkPageBreak(doc, requiredSpace = 100) {
  const remainingSpace = doc.page.height - doc.y - doc.page.margins.bottom;
  if (remainingSpace < requiredSpace) {
    doc.addPage();
  }
}
```

## HTTP Response Pattern

```javascript
// In route handler:
router.get('/reports/:structureId/performance', authenticate, async (req, res) => {
  const metrics = await ilpaService.calculatePerformanceMetrics(req.params.structureId);
  const pdfBuffer = await pdfGenerator.generatePerformanceReportPDF(metrics);

  res.setHeader('Content-Type', 'application/pdf');
  res.setHeader('Content-Disposition', `attachment; filename="performance-report-${req.params.structureId}.pdf"`);
  res.setHeader('Content-Length', pdfBuffer.length);
  res.send(pdfBuffer);
});
```
